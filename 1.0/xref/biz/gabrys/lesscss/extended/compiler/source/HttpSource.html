13  package biz.gabrys.lesscss.extended.compiler.source;
14  
15  import java.io.IOException;
16  import java.net.HttpURLConnection;
17  import java.net.MalformedURLException;
18  import java.net.URL;
19  import java.nio.charset.Charset;
20  import java.util.Arrays;
21  import java.util.Collection;
22  import java.util.Date;
23  import java.util.HashSet;
24  
25  import org.apache.commons.io.IOUtils;
26  
27  /**
28   * Represents a <a href="http://lesscss.org/">Less</a> source file accessible via
29   * <a href="http://www.w3.org/Protocols/">HTTP</a> or HTTPS.
30   * @since 1.0
31   */
32  public class HttpSource implements LessSource {
33  
34      private static final Collection<Integer> REDIRECT_CODES = new HashSet<Integer>(
35              Arrays.asList(HttpURLConnection.HTTP_MOVED_PERM, HttpURLConnection.HTTP_MOVED_TEMP, HttpURLConnection.HTTP_SEE_OTHER));
36  
37      private final URL url;
38      private String encoding;
39      private Date lastModificationDate;
40  
41      /**
42       * Constructs a new instance of the {@link HttpSource} and sets URL of the source file.
43       * @param url the source file URL.
44       * @since 1.0
45       */
46      public HttpSource(final URL url) {
47          this.url = url;
48      }
49  
50      public String getPath() {
51          return url.toString();
52      }
53  
54      /**
55       * {@inheritDoc} Before the first {@link #getContent()} call returns current source encoding. After the first
56       * {@link #getContent()} call always returns the source encoding read while downloading the source contents.
57       * @since 1.0
58       */
59      public String getEncoding() {
60          if (encoding != null) {
61              return encoding;
62          }
63  
64          final HttpURLConnection connection = makeConnection(false);
65          return getEncodingFromContentType(connection.getContentType());
66      }
67  
68      private static String getEncodingFromContentType(final String contentType) {
69          if (contentType == null || !contentType.contains("charset=")) {
70              return Charset.defaultCharset().toString();
71          }
72          return contentType.substring(contentType.lastIndexOf("=") + 1);
73      }
74  
75      public String getContent() {
76          final HttpURLConnection connection = makeConnection(true);
77          String content;
78          try {
79              content = IOUtils.toString(connection.getInputStream(), encoding);
80          } catch (final IOException e) {
81              throw new SourceException(e);
82          }
83          encoding = getEncodingFromContentType(connection.getContentType());
84          lastModificationDate = getModificationDate(connection.getLastModified());
85          return content;
86      }
87  
88      /**
89       * {@inheritDoc} Before the first {@link #getContent()} call returns current source modification time. After the
90       * first {@link #getContent()} call always returns the source modification time read while downloading the source
91       * contents.
92       * @since 1.0
93       */
94      public Date getLastModificationDate() {
95          if (lastModificationDate != null) {
96              return (Date) lastModificationDate.clone();
97          }
98  
99          final HttpURLConnection connection = makeConnection(false);
100         return getModificationDate(connection.getLastModified());
101     }
102 
103     private static Date getModificationDate(final long modificationDate) {
104         if (modificationDate == 0) {
105             return new Date();
106         }
107         return new Date(modificationDate);
108     }
109 
110     private HttpURLConnection makeConnection(final boolean fetchResponseBody) {
111         HttpURLConnection connection;
112         int responseCode;
113         URL resourceUrl = url;
114         while (true) {
115             try {
116                 connection = (HttpURLConnection) resourceUrl.openConnection();
117                 connection.setRequestMethod(fetchResponseBody ? "GET" : "HEAD");
118                 responseCode = connection.getResponseCode();
119             } catch (final IOException e) {
120                 throw new SourceException(e);
121             }
122 
123             if (!REDIRECT_CODES.contains(responseCode)) {
124                 break;
125             }
126             final String location = connection.getHeaderField("Location");
127             try {
128                 resourceUrl = new URL(location);
129             } catch (final MalformedURLException e) {
130                 throw new SourceException("Invalid Location header: " + location, e);
131             }
132         }
133         if (responseCode != HttpURLConnection.HTTP_OK) {
134             throw new SourceException(String.format("Cannot fetch content form network (error code: %s)", responseCode));
135         }
136         return connection;
137     }
138 }
