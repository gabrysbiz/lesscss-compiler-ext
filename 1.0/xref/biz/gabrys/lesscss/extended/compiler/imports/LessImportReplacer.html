15  /**
16   * Responsible for replacing import operations in <a href="http://lesscss.org/">Less</a> source code with new values.
17   * @since 1.0
18   */
19  public interface LessImportReplacer {
20  
21      /**
22       * Replaces a path of the import operation in source code with a new value.
23       * @param sourceCode the source code.
24       * @param operation the import operation.
25       * @param importPath the new value of the import operation path.
26       * @return a source code with replaced path in the import operation.
27       * @since 1.0
28       */
29      String replace(String sourceCode, LessImportOperation operation, String importPath);
30  }
