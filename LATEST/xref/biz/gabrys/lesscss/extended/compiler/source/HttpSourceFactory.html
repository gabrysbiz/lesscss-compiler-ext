13  package biz.gabrys.lesscss.extended.compiler.source;
14  
15  import java.net.MalformedURLException;
16  import java.net.URI;
17  import java.net.URISyntaxException;
18  
19  import org.apache.commons.lang.StringUtils;
20  
21  /**
22   * Responsible for creating new instances of the {@link HttpSource}.
23   * @since 1.0
24   */
25  public class HttpSourceFactory implements ConcreteSourceFactory {
26  
27      /**
28       * Constructs a new instance.
29       * @since 1.0
30       */
31      public HttpSourceFactory() {
32          // do nothing
33      }
34  
35      public LessSource createAbsoluteSource(final LessSource source, final String importAbsolutePath) {
36          try {
37              final URI importUri = new URI(importAbsolutePath).normalize();
38              return new HttpSource(importUri.toURL());
39          } catch (final URISyntaxException e) {
40              throw new SourceFactoryException("Cannot normalize URL", e);
41          } catch (final MalformedURLException e) {
42              throw new SourceFactoryException("Cannot create relative URL", e);
43          }
44      }
45  
46      public LessSource createRelativeSource(final LessSource source, final String importRelativePath) {
47          try {
48              final String sourcePath = source.getPath();
49              final String parentPath = sourcePath.substring(0, sourcePath.lastIndexOf('/'));
50              final URI importUri = new URI(parentPath + '/' + importRelativePath).normalize();
51              return new HttpSource(importUri.toURL());
52          } catch (final URISyntaxException e) {
53              throw new SourceFactoryException("Cannot normalize URL", e);
54          } catch (final MalformedURLException e) {
55              throw new SourceFactoryException("Cannot create relative URL", e);
56          }
57      }
58  
59      public boolean isAbsolutePath(final String path) {
60          if (StringUtils.isEmpty(path)) {
61              return false;
62          }
63          return path.startsWith("http://") || path.startsWith("https://");
64      }
65  }
